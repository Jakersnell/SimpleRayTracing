#![allow(unused)]

use std::{fs::File, io::BufWriter};

use image::{ImageBuffer, ImageOutputFormat, Rgb, RgbImage};
use lights::pointlight::PointLight;
use materials::{matte::Matte, mirror::Mirror};
use math::vector::Vector3;
use objects::{object::Object, sphere::Sphere};
use rendercore::{camera::Camera, world::World};
use utils::color::Color;

pub mod rendercore {
    pub mod camera;
    pub mod render;
    pub mod world;
}
pub mod lights {
    pub mod pointlight;
}
pub mod materials {
    pub mod material;
    pub mod matte;
    pub mod mirror;
    pub mod brdf {
        pub mod lambertian;
        pub mod specular;
    }
}
pub mod math {
    pub mod ray;
    pub mod utils;
    pub mod vector;
}
pub mod objects {
    pub mod cube;
    pub mod object;
    pub mod plane;
    pub mod sphere;
}
pub mod utils {
    pub mod color;
}

fn main() -> std::io::Result<()> {
    // Instantiate settings.
    const HEIGHT: u32 = 800;
    const WIDTH: u32 = 800;
    const SAMPLES: u64 = 10;
    const MAX_DEPTH: u8 = 2;
    let filename = "test.png".to_owned();
    // Initilize constructs.
    let mut world = World::new();
    let mut camera = Camera::new(HEIGHT, WIDTH, 1.0, 1.0, 1.0);
    // Initialize physical objects.
    let mirror_sphere = Object::new(
        Box::new(Sphere::new(Vector3::new(-1.0, 2.0, -10.0), 5.0)),
        Box::new(Matte::new(Color::new(0.0, 1.0, 0.0), 0.5)),
    );
    world.add_object(mirror_sphere);
    let matte_green_sphere = Object::new(
        Box::new(Sphere::new(Vector3::new(5.0, 3.0, -10.0), 3.0)),
        Box::new(Matte::new(Color::new(0.0, 1.0, 0.0), 0.5)),
    );
    world.add_object(matte_green_sphere);
    // Initialize lights.
    let pointlight = PointLight::new(Vector3::new(-10.0, -10.0, 0.0), Color::white(), 10.0);
    world.add_light(pointlight);
    // Given camera and world, render the scene, return render as Vec<u8>.
    let byte_buf = rendercore::render::render_pixels(
        &camera,
        &world,
        f64::EPSILON,
        f64::MAX,
        SAMPLES,
        MAX_DEPTH,
    );
    // Iterate over pixels, write raw pixel to each pixel.
    let mut pixel_iter = byte_buf.into_iter();
    let mut image = RgbImage::new(WIDTH, HEIGHT);
    for x in 0..WIDTH {
        for y in 0..HEIGHT {
            if let Some(pixel) = pixel_iter.next() {
                image.put_pixel(x, y, Rgb::from(pixel))
            } else {
                panic!()
            }
        }
    }
    // Create output filestream.
    let file = File::create(format!("output/{}", filename))?;
    // Write to image
    image.write_to(&mut BufWriter::new(file), ImageOutputFormat::Png);
    Ok(())
}
